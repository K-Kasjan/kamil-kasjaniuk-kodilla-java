SELECT I.SUMMARY, I.DESCRIPTION, L.NAME AS LISTS_NAME
FROM issues I JOIN issueslists L ON I.ISSUESLIST_ID = L.ID;

SELECT i.SUMMARY, i.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM issues i JOIN users u on i.USER_ID_ASSIGNEDTO = u.ID;

SELECT u.LASTNAME, u.FIRSTNAME, COUNT(i.SUMMARY) AS TASKS_COUNT
FROM users u join issues i on u.ID = i.USER_ID_ASSIGNEDTO
GROUP BY u.LASTNAME, u.FIRSTNAME;
